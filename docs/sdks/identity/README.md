    ClientRequestId = "71010d88-d0e7-4a24-9297-d1be6fefde81",
    Items = new List<IdentityItem>() {
        new IdentityItem() {
            ClientCustomerId = "e0f78bc2-f748-4eda-9d29-d756844507fc",
            DeviceId = "bf9ea15d-7dfa-4bb4-a64c-6c26b53472fc",
            PhoneNumber = "2001001695",
